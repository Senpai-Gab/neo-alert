import { neoAlert } from 'neo-alert';

function SaveButton() {
  const handleSave = async () => {
    const id = neoAlert.info('Saving…', { duration: 0 });
    try {
      await saveData();
      neoAlert.dismiss(id);
      neoAlert.success('Saved!', { message: 'All changes persisted.' });
    } catch (err) {
      neoAlert.dismiss(id);
      neoAlert.error('Save failed', { message: err.message });
    }
  };
  return <button onClick={handleSave}>Save</button>;
}

// Promise shorthand
const handleUpload = () => {
  neoAlert.promise(uploadFile(file), {
    loading: 'Uploading…',
    success: (data) => `Uploaded ${data.name}!`,
    error:   (err)  => `Failed: ${err.message}`,
  });
};
// main.js — register globally
import { createApp } from 'vue';
import { neoAlert } from 'neo-alert';
import App from './App.vue';

const app = createApp(App);
app.config.globalProperties.$toast = neoAlert;
app.mount('#app');

// In any .vue component:
<script setup>
import { getCurrentInstance } from 'vue';
const { proxy } = getCurrentInstance();
const notify = () => proxy.$toast.success('Vue toast!');
</script>

API Reference

      
Complete API

      
All methods, options, and types documented in one place.


      

        

          
neoAlert(options)

          
Core call with full options object. Returns the unique toast ID.

          

            

              type
              string
              success | error | warning | info
            

            

              title
              string
              Headline text (required)
            

            

              message
              string?
              Optional body description
            

            

              position
              string?
              top-right | top-left | bottom-right | bottom-left
            

            

              duration
              number?
              Auto-dismiss ms. 0 = persistent
            

            

              showProgress
              boolean?
              Show countdown progress bar
            

            

              pauseOnHover
              boolean?
              Freeze timer on mouse hover
            

            

              draggable
              boolean?
              Enable drag-to-dismiss
            

            

              theme
              object?
              Custom visual overrides (see customizer)
            

            

              onDismiss
              fn?
              Callback fired when toast closes
            

          

        

        

          
neoAlert.promise(p, msgs)

          
Wrap a Promise — shows spinner, auto-resolves to success or error.

          

            

              p
              Promise
              The async operation to track
            

            

              msgs.loading
              string
              Shown while loading
            

            

              msgs.success
              string|fn
              On resolve: string or (data)=>string
            

            

              msgs.error
              string|fn
              On reject: string or (err)=>string
            

          

        

        

          
neoAlert.configure(opts)

          
Set global defaults once at app start. All toasts inherit these.

          

            

              position
              string
              Default position
            

            

              duration
              number
              Default duration in ms
            

            

              maxToasts
              number
              Max simultaneous (default: 6)
            

            

              pauseOnHover
              boolean
              Global pause-on-hover toggle
            

            

              showProgress
              boolean
              Global progress bar toggle
            

            

              draggable
              boolean
              Global drag-to-dismiss toggle
            

          

        

        

          
neoAlert.dismiss(id)

          
Dismiss a specific toast programmatically using its returned ID.

          

            

              id
              string
              ID returned from any neoAlert() call
            

          

          

          
neoAlert.clearAll()

          
Dismiss every active toast with smooth exit animations.

          

          
neoAlert.version

          
Returns the current version string: "1.0.0"
          

        

      

    


    


    
    

      
Features

      
Everything you need

      

      

        

          

          
Glassmorphism

          
Frosted-glass UI with backdrop-filter blur, layered shadows, and translucent backgrounds.

        

        

          

          
Smart Stacking

          
Intelligent queue — toasts stack smoothly, oldest is removed automatically when the limit is reached.

        

        

          

          
Auto Dismiss + Pause

          
Progress bar counts down to auto-dismiss. Hover-pause freezes the timer exactly where it stopped.

        

        

          

          
Promise API

          
Track any async operation — loading spinner auto-transitions to success or error on completion.

        

        

          

          
Drag to Dismiss

          
Pointer and touch drag with opacity feedback. 80px threshold before committing; snaps back otherwise.

        

        

          

          
Full Theming

          
Override any property per-toast: background, border, colors, radius, blur, shadow. 6 presets included.
